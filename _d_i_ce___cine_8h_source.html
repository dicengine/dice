149 class DICE_LIB_DLL_EXPORT
150 Cine_Header
151 {
152 public:
157  Cine_Header(const std::string & file_name,
158  const cine_file_header & header,
159  const bitmap_info_header & bitmap_header):
160  header_(header),
161  bitmap_header_(bitmap_header),
162  file_name_(file_name),
163  bit_depth_(NO_SUCH_BIT_DEPTH){
164  image_offsets_ = new int64_t[header_.ImageCount];
165  // set the bit depth of the images
166  int_t bit_depth = (bitmap_header_.biSizeImage * 8) / (bitmap_header_.biWidth * bitmap_header_.biHeight);
167  if(bit_depth==8) bit_depth_=BIT_DEPTH_8;
168  else if (bit_depth==16) bit_depth_=BIT_DEPTH_16;
169  else if (bit_depth==10) bit_depth_=BIT_DEPTH_10_PACKED;
170  else{
171  TEUCHOS_TEST_FOR_EXCEPTION(true,std::runtime_error,"Error, invalid bit depth .cine");
172  }
173  };
175  virtual ~Cine_Header(){
176  delete[] image_offsets_;
177  };
179  cine_file_header header_;
181  bitmap_info_header bitmap_header_;
183  int64_t * image_offsets_;
185  std::string file_name_;
187  Bit_Depth bit_depth_;
188 };
