119  Matrix(std::initializer_list<std::initializer_list<Type> > list):
120  Matrix<Type,Rows,Cols>(){
121  // check if the default constructor was called with no dimensions and fix
122  assert(list.size()<=MAX_MATRIX_DIM);
123  assert(list.begin()->size()<=MAX_MATRIX_DIM);
124  if(rows_==MAX_MATRIX_DIM&&cols_==MAX_MATRIX_DIM){
125  rows_ = list.size();
126  cols_ = list.begin()->size();

128  assert(list.size()==rows_);
129  assert(list.begin()->size()!=0);
130  assert(list.begin()->size()==cols_);
131  std::vector<std::initializer_list<Type> > set_vec = list;
132  std::vector<std::vector<Type> > v;
133  for (auto i = set_vec.begin(); i != set_vec.end(); i++)
134  v.push_back(std::vector<Type>(*i));
135  for (size_t i = 0; i < rows_; i++)
136  for(size_t j=0;j<cols_;++j)

255  template <size_t RRows, size_t RCols>
256  Matrix<Type,Rows,Cols> operator-(const Matrix<Type,RRows,RCols> & rhs)const{
257  validate_dimensions(rhs);

259  for(size_t i=0;i<rows_;++i)

266  template <typename RType,size_t RRows,size_t RCols>
267  Matrix<Type,Rows,RCols> operator*(const Matrix<RType,RRows,RCols> & rhs)const{
268  static_assert(std::is_same<Type,RType>::value,"scalar type must be the same to multiply matrices");
269  TEUCHOS_TEST_FOR_EXCEPTION(cols_!=rhs.rows(),std::runtime_error,"matrix dimensions are not compatible");
270  Matrix<Type,Rows,RCols> ans(rows_,rhs.cols());
271  const size_t rhs_cols = rhs.cols();
272  const size_t rhs_rows = rhs.rows();
273  for(size_t i=0;i<rows_;++i){
274  for(size_t j=0;j<rhs_cols;++j){
275  for(size_t k=0;k<rhs_rows;++k){
276  ans(i,j) += (*this)(i,k)*rhs[k*rhs_cols+j];
277  }
278  }
279  }
280  return ans;
281  };

332  // always has to be scalar type because of lapack
333  Matrix<scalar_t,Rows,Cols> inv(){
334  TEUCHOS_TEST_FOR_EXCEPTION(rows_!=cols_,std::runtime_error,"matrix must be square");
335  Matrix<scalar_t,Rows,Cols> inverse;
336  inverse.copy(*this); // lapack only works on scalar type not integers, copy needed to cast values
337  int_t info = 0;
338  std::fill(int_work_.begin(), int_work_.end(), 0);
339  lapack_.GETRF(rows_,rows_,inverse.data(),
340  rows_,&int_work_[0],&info);
341  std::fill(scalar_work_.begin(), scalar_work_.end(), 0);
342  lapack_.GETRI(rows_,inverse.data(),rows_,&int_work_[0],
343  &scalar_work_[0],rows_*rows_,&info);
344  return inverse;
345  }

347  scalar_t condition_number(){
348  TEUCHOS_TEST_FOR_EXCEPTION(rows_!=cols_,std::runtime_error,"matrix must be square");
349  Matrix<scalar_t,Rows,Cols> scalar_mat;
350  scalar_mat.copy(*this); // lapack only works on scalar type not integers, copy needed to cast values
351  int_t info = 0;
352  std::fill(int_work_.begin(), int_work_.end(), 0);
353  lapack_.GETRF(rows_,rows_,scalar_mat.data(),
354  rows_,&int_work_[0],&info);
355  scalar_t rcond = 0.0;
356  scalar_t norm = one_norm(scalar_mat);
357  lapack_.GECON('1',rows_,scalar_mat.data(),rows_,norm,&rcond,
358  &scalar_work_[0],&int_work_[0],&info);
359  return rcond==0.0?0.0:1.0/rcond;
360  }

452 template <typename Type, size_t Rows, size_t Cols>
453 scalar_t one_norm(Matrix<Type,Rows,Cols> matrix){
454  scalar_t norm = 0.0;
455  Vector<Type,Cols> col_totals;
456  for(size_t i=0;i<matrix.rows();++i)
457  for(size_t j=0;j<matrix.cols();++j)
458  col_totals(j) += std::abs(matrix(i,j));
459  for(size_t j=0;j<matrix.cols();++j)
460  if(col_totals[j] > norm) norm = col_totals[j];
461  return norm;
462 };
