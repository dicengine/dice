82  Objective(Schema * schema, const int_t correlation_point_global_id):
83  schema_(schema),
84  correlation_point_global_id_(correlation_point_global_id)
85 {
86  if(correlation_point_global_id==-1)return;
87  assert(schema_->is_initialized());
88  // create the refSubset as member data
89  // check to see if the schema has multishapes:
90  if((*schema_->conformal_subset_defs()).find(correlation_point_global_id_)!=(*schema_->conformal_subset_defs()).end()){
91  subset_ = Teuchos::rcp(new Subset(static_cast<int_t>(global_field_value(DICe::field_enums::SUBSET_COORDINATES_X_FS)),
92  static_cast<int_t>(global_field_value(DICe::field_enums::SUBSET_COORDINATES_Y_FS)),
93  (*schema_->conformal_subset_defs()).find(correlation_point_global_id_)->second));
94  }
95  // otherwise build up the subsets from x/y and w/h:
96  else{
97  assert(schema_->subset_dim()>0);
98  subset_ = Teuchos::rcp(new Subset(static_cast<int_t>(global_field_value(DICe::field_enums::SUBSET_COORDINATES_X_FS)),
99  static_cast<int_t>(global_field_value(DICe::field_enums::SUBSET_COORDINATES_Y_FS)),
100  schema_->subset_dim(),schema_->subset_dim()));
101  }
102  assert(schema_->ref_img()!=Teuchos::null);
103  subset_->initialize(schema_->ref_img());
104 }
105 
110  Objective(Schema * schema,
111 const int_t x,
112 const int_t y):
113  schema_(schema),
114  correlation_point_global_id_(-1)
115 {
116  // create the refSubset as member data from x/y and w/h:
117  assert(schema_->is_initialized());
118  assert(schema_->subset_dim()>0);
119  assert(x>=0&&x<schema_->ref_img()->width());
120  assert(y>=0&&y<schema_->ref_img()->height());
121  assert(schema_->ref_img()!=Teuchos::null);
122  subset_ = Teuchos::rcp(new Subset(x,y,schema_->subset_dim(),schema_->subset_dim()));
123  subset_->initialize(schema_->ref_img());
124 }
