181  struct DICE_LIB_DLL_EXPORT Camera_Info{
182  // array of camera intrinsics (the distortion coeffs are not stored in the intrinsic array)
183  // openCV_DIS - (cx cy fx fy k1 k2 p1 p2 [k3] [k4 k5 k6] [s1 s2 s3 s4] [tx ty]) 4,5,8,12 or 14 distortion coef
184  // Vic3D_DIS - (cx cy fx fy fs k1 k2 k3) 3 distortion coef (k1r1 k2r2 k3r3)
185  // generic1 - (cx cy fx fy fs k1 k2 k3) 3 distortion coef (k1r1 k2r2 k3r3)
186  // generic2 - (cx cy fx fy fs k1 k2 k3) 3 distortion coef (k1r2 k2r4 k3r6)
187  // generic3 - (cx cy fx fy fs k1 k2 k3) 3 distortion coef (k1r3 k2r5 k3r7)
188  std::vector<scalar_t> intrinsics_;
189  // extrinsic rotation_matrix [R] matrix transforming from the world coordinates to the camera coordinates
190  // this matrix is the identity matrix if the world coordinates are the camera coordinates
191  //
192  // a camera always stores its extrinsic information as a rotation matrix and three translations
193  // if the euler angles are specified, they are converted to a rotation matrix, the eulers are not stored
194  Matrix<scalar_t,3> rotation_matrix_;
195  // extrinsic translations x y and z (these are optional and zero if the world coordinates are the camera coordinates)
196  scalar_t tx_;
197  scalar_t ty_;
198  scalar_t tz_;
199  // lens distortion model
200  Lens_Distortion_Model lens_distortion_model_;
201  // image dimensions in pixels
202  int_t image_width_;
203  int_t image_height_;
204  // optional
205  int_t pixel_depth_;
206  // camera information
207  // identifier
208  std::string id_;
209  // lens type
210  std::string lens_;
211  // comments about the camera
212  std::string comments_;
213  // constructor
214  Camera_Info():
215  intrinsics_(MAX_CAM_INTRINSIC_PARAM,0),
216  rotation_matrix_(Matrix<scalar_t,3>::identity()),
217  tx_(0),
218  ty_(0),
219  tz_(0),
220  lens_distortion_model_(NO_LENS_DISTORTION),
221  image_width_(-1),
222  image_height_(-1),
223  pixel_depth_(-1){};
224  // convert euler angles to rotation matrix
225  static Matrix<scalar_t,3> eulers_to_rotation_matrix(const scalar_t & alpha,
226  const scalar_t & beta,
227  const scalar_t & gamma);
228  // convert euler angles to rotation matrix
229  static void rotation_matrix_to_eulers(const Matrix<scalar_t,3> & R,
230  scalar_t & alpha,
231  scalar_t & beta,
232  scalar_t & gamma);
233  // return the euler angles for this camera
234  void eulers(scalar_t & alpha,
235  scalar_t & beta,
236  scalar_t & gamma) const {
237  rotation_matrix_to_eulers(rotation_matrix_,alpha,beta,gamma);
238  }
239  // convert euler angles to rotation matrix partial derivatives
240  static void eulers_to_rotation_matrix_partials(const scalar_t & alpha,
241  const scalar_t & beta,
242  const scalar_t & gamma,
243  Matrix<scalar_t,3,4> & R_dx,
244  Matrix<scalar_t,3,4> & R_dy,
245  Matrix<scalar_t,3,4> & R_dz);
246  // set the rotation matrix using the euler angles alpha, beta, and gamma
247  void set_rotation_matrix(const scalar_t & alpha,
248  const scalar_t & beta,
249  const scalar_t & gamma);
250  // set the intrinsic vector from an already constructed vector
251  void set_intrinsics(const std::vector<scalar_t> & in){
252  TEUCHOS_TEST_FOR_EXCEPTION(in.size()!=MAX_CAM_INTRINSIC_PARAM,std::runtime_error,"");
253  for(size_t i=0;i<intrinsics_.size();++i) intrinsics_[i] = in[i];
254  }
255  // helper function to set the translations
256  void set_extrinsic_translations(const scalar_t & tx, const scalar_t & ty, const scalar_t & tz){
257  tx_ = tx; ty_ = ty; tz_ = tz;
258  }
259  // return true if the camera has all the parameters set needed to be used for analysis
260  bool is_valid();
261  // same as is_valid() only throws an exception if not valid
262  void check_valid()const;
263  // clears all the parameter values in a camera
264  void clear();
265  // difference with another camera info
266  scalar_t diff(const Camera_Info & rhs) const;
267  // comparison operator
268  DICE_LIB_DLL_EXPORT
269  friend bool operator==(const Camera_Info & lhs,const Camera_Info & rhs);
270  // comparison operator
271  DICE_LIB_DLL_EXPORT
272  friend bool operator!=(const Camera_Info & lhs,const Camera_Info & rhs){
273  return !(lhs==rhs);
274  }
275  // overaload the ostream operator for a camera info class
276  DICE_LIB_DLL_EXPORT
277  friend std::ostream & operator<<(std::ostream & os, const Camera_Info & info);
