140  scalar_t diff(const Camera_System & rhs) const{
141  TEUCHOS_TEST_FOR_EXCEPTION(rhs.num_cameras()!=num_cameras(),std::runtime_error,"");
142  scalar_t diff_ = 0.0;
143  for(size_t i=0;i<num_cameras();++i)
144  diff_ += camera(i)->diff(*(rhs.camera(i).get()));
145  return diff_;
146  }
