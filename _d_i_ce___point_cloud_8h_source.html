116  {
117  const T d0=p1[0]-pts[idx_p2].x;
118  const T d1=p1[1]-pts[idx_p2].y;
119  const T d2=p1[2]-pts[idx_p2].z;

125  inline T kdtree_get_pt(const size_t idx, int dim) const

130  }
