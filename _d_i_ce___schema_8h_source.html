547  DEBUG_MSG("Saving off solution nm1 for subset (global id) " << global_id);
548  for(int_t i=0;i<DICe::field_enums::num_fields_defined;++i){
549  DICe::field_enums::Field_Spec fs_nm1 = DICe::field_enums::fs_spec_vec[i];

551  // build up a field spec for a no state field corresponding to the nm1 field
